/**
 * Engagement Lab Website
 * 
 * Project Model
 * @module project
 * @author Johnny Richardson
 * 
 * For field docs: http://keystonejs.com/docs/database/
 *
 * ==========
 */

var keystone = require('keystone');
var Types = keystone.Field.Types;

/**
 * @module project
 * @constructor
 */
var Project = new keystone.List('Project');

/**
 * Model Fields
 * @main Project
 */
Project.add({
	name: { type: String, label: 'Project Name', required: true, index: true },
	byline: { type: Types.Markdown, label: 'Byline Description', initial: true, required: true },
	overview: { type: Types.Markdown, label: 'Project Overview', initial: true, required: true },
	highlights: { type: String, label: 'Key Features/Highlights', initial: true, required: true },
	header_image: { type: Types.CloudinaryImages, label: 'Header Image (large)', folder: 'research/home', autoCleanup : true },
	slug_image: { type: Types.CloudinaryImages, label: 'Project Images', folder: 'research/projects', autoCleanup : true },
	
	createdAt: { type: Date, default: Date.now }
});

/**
 * Model Registration
 */
Project.defaultSort = '-createdAt';
Project.defaultColumns = 'name, short_description, description, slug_image';
Project.register();
